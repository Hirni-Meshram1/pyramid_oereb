[docs]    def to_extract(self):
        """
        Returns a dictionary with all available values needed for the extract.
        :return: Dictionary with values for the extract.
        :rtype: dict
        """
        extract = dict()
        for key in [
            'content',
            'subtopic',
            'additional_topic',
            'type_code',
            'type_code_list',
            'legal_state',
            'area',
            'part_in_percent',
            'symbol'
        ]:
            value = getattr(self, key)
            if value:
                extract[key] = value
        for key in [
            'geometries',
            'documents'
        ]:
            records = getattr(self, key)
            if records and len(records) > 0:
                extract[key] = [record.to_extract() for record in records]
        for key in [
            'theme',
            'responsible_office'
        ]:
            record = getattr(self, key)
            if record:
                extract[key] = record.to_extract()
        key = 'view_service'
        record = getattr(self, key)
        if record:
            extract[key] = record.to_extract(type_code=self.type_code)
        return extract
